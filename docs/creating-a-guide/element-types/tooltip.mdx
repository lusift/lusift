Creating Tooltip element

```
type Tooltip = {
    index: number;
    type: 'tooltip';
    data: TooltipData;
    target: Target;
    actions: StepActions;
    styleProps: object;
}

type TooltipData = {
    bodyContent: BodyContent;
    footerContent: FooterContent;
    placement: Placement;
    offset: [number, number];
    arrow: boolean;
    arrowSizeScale: number;
    scrollIntoView: boolean;
    backdrop: {
        disabled: boolean;
        color: string;
        opacity: string;
        stageGap: number;
        nextOnOverlayClick: boolean;
    };
    progressOn: {
        eventType: string;
        elementSelector: string;
        disabled: boolean;
    };
    progressBar: {
        disabled: boolean;
        styleProps: object;
    };
}

type BodyContent = string | Element | object | Function;
type FooterContent = string | Element | object | Function;

type Target = {
    path: {
        value: string;
        comparator: PathComparator;
    };
    elementSelector: string;
}

type PathComparator =
    'is' |
    'contains' |
    'startsWith' |
    'endsWith' |
    'regex';


type Position = Placement | 'auto';
type Orientation = 'fixed' | 'auto';
type Placement = {
    position: Position;
    orientation: Orientation;
}

type Alignment = 'start' | 'end';
type Side = 'top' | 'right' | 'bottom' | 'left';
type AlignedPlacement = `${Side}-${Alignment}`;
```
